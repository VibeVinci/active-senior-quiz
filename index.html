function shuffle(a) {

  const b = [...a];

  for (let i = b.length - 1; i > 0; i--) {

    const j = Math.floor(Math.random() * (i + 1));

    [b[i], b[j]] = [b[j], b[i]];

  }

  return b;

}

}


// ── TTS ──

function useTTS() {

  const [speaking, setSpeaking] = useState(false);

  const sup = "speechSynthesis" in window;

  const speak = useCallback((text, rate = 0.85) => {

    if (!sup) return;

    window.speechSynthesis.cancel();

    const u = new SpeechSynthesisUtterance(text);

    u.lang = "zh-TW"; u.rate = rate; u.pitch = 1.05;

    u.onstart = () => setSpeaking(true);

    u.onend = () => setSpeaking(false);

    u.onerror = () => setSpeaking(false);

    window.speechSynthesis.speak(u);

  }, [sup]);

  const stop = useCallback(() => { if (sup) { window.speechSynthesis.cancel(); setSpeaking(false); } }, [sup]);

  useEffect(() => () => { if (sup) window.speechSynthesis.cancel(); }, [sup]);

  return { speak, stop, speaking, supported: sup };

}


// ── Audio ──

function useAudio() {

  const ref = useRef(null);

  function ctx() { if (!ref.current) ref.current = new (window.AudioContext || window.webkitAudioContext)(); return ref.current; }

  function beep(freq, dur, type = "sine", vol = 0.4) {

    try {

      const c = ctx(), o = c.createOscillator(), g = c.createGain();

      o.connect(g); g.connect(c.destination);

      o.type = type; o.frequency.value = freq;

      g.gain.setValueAtTime(vol, c.currentTime);

      g.gain.exponentialRampToValueAtTime(0.001, c.currentTime + dur);

      o.start(c.currentTime); o.stop(c.currentTime + dur);

    } catch (e) {}

  }

  return {

    playCorrect: () => { beep(520,.12); setTimeout(()=>beep(660,.18),110); setTimeout(()=>beep(880,.25),230); },

    playWrong:   () => { beep(300,.15,"sawtooth",.35); setTimeout(()=>beep(220,.25,"sawtooth",.3),140); },

    playTick:    () => beep(440,.06,"square",.15),

    playTimeout: () => beep(200,.3,"sawtooth",.35),

    playStart:   () => { beep(440,.1); setTimeout(()=>beep(550,.1),100); setTimeout(()=>beep(660,.2),200); },

    playFinish:  () => { [440,550,660,880].forEach((f,i)=>setTimeout(()=>beep(f,.18),i*100)); },

  };

}

          ✨ 開始挑戰！

        </button>

        <p style={{ textAlign:"center", fontSize:"17px", color:"#A78BFA", marginTop:"14px", fontWeight:"600" }}>🌟 每次題目不同，挑戰自己每天進步！</p>

      </div>

    </div>

  );

// ════════════════════════════════════════════ QUIZ

function QuizPage({ name, questions, onFinish }) {

  const [qIdx, setQIdx] = useState(0);

  const [selected, setSelected] = useState(null);

  const [confirmed, setConfirmed] = useState(false);

  const [timeLeft, setTimeLeft] = useState(TOTAL_TIME);

  const scoresRef = useRef([]);

  const timerRef = useRef(null);

  const confirmedRef = useRef(false);

  const tts = useTTS(); const audio = useAudio();

  const cur = questions[qIdx];


  useEffect(() => {

    confirmedRef.current = false;

    setSelected(null); setConfirmed(false); setTimeLeft(TOTAL_TIME);

    const opts = cur.opts.map((o,i) => `${String.fromCharCode(65+i)}，${o}`).join("。");

    setTimeout(() => tts.speak(`第${qIdx+1}題。${cur.q}　選項：${opts}`), 300);

  }, [qIdx]);


  useEffect(() => {

    if (confirmed) return;

    timerRef.current = setInterval(() => {

      setTimeLeft(t => {

        if (t <= 4 && t > 1) audio.playTick();

        if (t <= 1) { clearInterval(timerRef.current); if (!confirmedRef.current) doConfirm(null, true); return 0; }

        return t - 1;

      });

    }, 1000);

    return () => clearInterval(timerRef.current);

  }, [qIdx, confirmed]);


  function doConfirm(sel, timeout = false) {

    if (confirmedRef.current) return;

    confirmedRef.current = true;

    clearInterval(timerRef.current); tts.stop();

    const ok = sel === cur.ans;

    scoresRef.current = [...scoresRef.current, ok];

    setSelected(sel); setConfirmed(true);

    if (timeout) { audio.playTimeout(); setTimeout(() => tts.speak(`時間到！正確答案是：${cur.opts[cur.ans]}。${cur.explanation}`), 300); }

    else if (ok) { audio.playCorrect(); setTimeout(() => tts.speak(`答對了！太棒了！${cur.explanation}`), 200); }

    else { audio.playWrong(); setTimeout(() => tts.speak(`這次差一點！正確答案是：${cur.opts[cur.ans]}。${cur.explanation}`), 200); }

  }


  function handleNext() {

    tts.stop();

    if (qIdx + 1 >= questions.length) {

      audio.playFinish();

      const s = scoresRef.current;

      setTimeout(() => tts.speak(`恭喜完成！您答對了${s.filter(Boolean).length}題，共${s.length}題。`), 400);

      onFinish(s);

    } else { setQIdx(q => q + 1); }

  }


  function optBg(i) { if (!confirmed) return selected===i?"#FEF3C7":"#fff"; if (i===cur.ans) return "#D1FAE5"; if (i===selected) return "#FEE2E2"; return "#F9FAFB"; }

  function optBorder(i) { if (!confirmed) return selected===i?"3px solid #F59E0B":"2px solid #FCD34D"; if (i===cur.ans) return "3px solid #10B981"; if (i===selected) return "3px solid #EF4444"; return "2px solid #E5E7EB"; }

  function optIcon(i) { if (!confirmed) return null; if (i===cur.ans) return "✅"; if (i===selected&&selected!==cur.ans) return "❌"; return null; }


  const tc = timeLeft>10?"#10B981":timeLeft>5?"#F59E0B":"#EF4444";

  const fullText = `第${qIdx+1}題。${cur.q}　選項：${cur.opts.map((o,i)=>`${String.fromCharCode(65+i)}，${o}`).join("。")}`;


  return (

    <div style={S.page}>

      <div style={{ maxWidth:"540px", width:"100%", display:"flex", alignItems:"center", justifyContent:"space-between", marginBottom:"12px" }}>

        <div style={{ fontSize:"19px", fontWeight:"800", color:"#D97706" }}>🧠 健康防詐金頭腦</div>

        <div style={{ fontSize:"19px", fontWeight:"700", color:"#92400E", background:"#FEF3C7", borderRadius:"12px", padding:"6px 14px" }}>第 {qIdx+1} / {questions.length} 題</div>

      </div>


      <div style={{ maxWidth:"540px", width:"100%", height:"13px", background:"#FDE68A", borderRadius:"99px", marginBottom:"14px", overflow:"hidden" }}>

        <div style={{ height:"100%", width:`${(qIdx/questions.length)*100}%`, background:"linear-gradient(90deg,#F59E0B,#EF4444)", borderRadius:"99px", transition:"width 0.4s" }} />

      </div>


      <div style={{ ...S.card, marginBottom:"12px" }}>

        <div style={{ display:"flex", alignItems:"center", justifyContent:"space-between", marginBottom:"14px" }}>

          <div style={{ width:"56px", height:"56px", borderRadius:"50%", background:"#FEF3C7", border:`4px solid ${tc}`, display:"flex", flexDirection:"column", alignItems:"center", justifyContent:"center" }}>

            <span style={{ fontSize:"21px", fontWeight:"900", color:tc, lineHeight:1 }}>{timeLeft}</span>

            <span style={{ fontSize:"11px", color:"#92400E", fontWeight:"600" }}>秒</span>

          </div>

          {tts.supported && (

            <div style={{ display:"flex", alignItems:"center", gap:"8px" }}>

              <span style={{ fontSize:"16px", color:"#92400E", fontWeight:"600" }}>再聽一次</span>

              <SpeakBtn text={fullText} speak={tts.speak} stop={tts.stop} speaking={tts.speaking} />

            </div>

          )}

        </div>

        {name && <p style={{ fontSize:"17px", color:"#A78BFA", fontWeight:"600", marginBottom:"6px" }}>👤 {name}，加油！</p>}

        <p style={{ fontSize:"22px", fontWeight:"800", color:"#78350F", lineHeight:"1.65" }}>{qIdx+1}. {cur.q}</p>

      </div>


      <div style={{ maxWidth:"540px", width:"100%", display:"flex", flexDirection:"column", gap:"10px", marginBottom:"12px" }}>

        {cur.opts.map((opt,i) => (

          <button key={i} onClick={() => { if (!confirmed) { setSelected(i); tts.speak(opt, 0.9); } }}

            style={S.btn({ width:"100%", minHeight:"66px", background:optBg(i), border:optBorder(i), fontSize:"20px", color:"#78350F", padding:"12px 18px", textAlign:"left", display:"flex", alignItems:"center", gap:"12px", cursor:confirmed?"default":"pointer", boxShadow:selected===i&&!confirmed?"0 4px 14px rgba(245,158,11,0.30)":"0 2px 8px rgba(0,0,0,0.06)" })}>

            <span style={{ width:"34px", height:"34px", borderRadius:"50%", background:selected===i&&!confirmed?"#F59E0B":"#FEF3C7", color:selected===i&&!confirmed?"#fff":"#D97706", display:"flex", alignItems:"center", justifyContent:"center", fontWeight:"900", fontSize:"17px", flexShrink:0 }}>

              {String.fromCharCode(65+i)}

            </span>

            <span style={{ flex:1 }}>{opt}</span>

            {optIcon(i) && <span style={{ fontSize:"22px" }}>{optIcon(i)}</span>}

          </button>

        ))}

      </div>


      {!confirmed ? (

        <div style={{ maxWidth:"540px", width:"100%" }}>

          <button onClick={() => selected !== null && doConfirm(selected)} disabled={selected===null}

            style={S.btn({ width:"100%", height:"66px", background:selected!==null?"linear-gradient(135deg,#F59E0B,#EF4444)":"#D1D5DB", color:"#fff", fontSize:"23px", cursor:selected!==null?"pointer":"not-allowed", boxShadow:selected!==null?"0 6px 18px rgba(245,158,11,0.40)":"none" })}>

            {selected !== null ? "✅ 確認答案" : "請選擇一個答案"}

          </button>

        </div>

      ) : (

        <div style={{ maxWidth:"540px", width:"100%", display:"flex", flexDirection:"column", gap:"10px" }}>

          <div style={{ background:selected===cur.ans?"#D1FAE5":"#FEE2E2", borderRadius:"16px", padding:"14px 18px", fontSize:"21px", fontWeight:"800", color:selected===cur.ans?"#065F46":"#991B1B", textAlign:"center" }}>

            {selected===cur.ans ? "🎉 答對了！太棒了！" : selected===null ? `⏰ 時間到！正確答案是「${cur.opts[cur.ans]}」` : `😊 差一點！正確答案是「${cur.opts[cur.ans]}」`}

          </div>

          <div style={{ background:"linear-gradient(135deg,#FFFBEB,#FEF3C7)", border:"2px solid #F59E0B", borderRadius:"18px", padding:"16px 18px", display:"flex", gap:"12px", alignItems:"flex-start" }}>

            <span style={{ fontSize:"28px", flexShrink:0 }}>💡</span>

            <div style={{ flex:1 }}>

              <p style={{ fontSize:"16px", fontWeight:"800", color:"#D97706", marginBottom:"6px" }}>溫馨小提醒</p>

              <p style={{ fontSize:"19px", color:"#78350F", fontWeight:"600", lineHeight:"1.7" }}>{cur.explanation}</p>

            </div>

            {tts.supported && <SpeakBtn text={cur.explanation} speak={tts.speak} stop={tts.stop} speaking={tts.speaking} />}

          </div>

          <button onClick={handleNext}

            style={S.btn({ width:"100%", height:"66px", background:"linear-gradient(135deg,#F59E0B,#FBBF24)", color:"#fff", fontSize:"23px", boxShadow:"0 6px 18px rgba(245,158,11,0.40)" })}>

            {qIdx+1 >= questions.length ? "🏆 看最終成績！" : "下一題 →"}

          </button>

        </div>

      )}

    </div>

  );

}


// ════════════════════════════════════════════ RESULT

function ResultPage({ name, scores, onRestart }) {

  const correct = scores.filter(Boolean).length;

  const total = scores.length;

  const pct = Math.round((correct / total) * 100);

  const grade = pct >= 80

    ? { label:"超級厲害！", icon:"🏆", color:"#D97706", msg:"您的健康防詐知識非常豐富，請繼續保持！" }

    : pct >= 60

    ? { label:"答得不錯！", icon:"🌟", color:"#059669", msg:"再多練習幾次，您會更厲害的！" }

    : { label:"繼續加油！", icon:"💪", color:"#7C3AED", msg:"別氣餒，多玩幾次就會越來越熟悉！" };

  const tts = useTTS();


  useEffect(() => {

    tts.speak(`${name?name+"，":""}您答對了${correct}題，共${total}題，答對率${pct}%。${grade.label}${grade.msg}`);

  }, []);


  return (

    <div style={{ ...S.page, justifyContent:"center" }}>

      <div style={{ ...S.card, maxWidth:"500px", textAlign:"center" }}>

        <div style={{ fontSize:"76px", marginBottom:"10px" }}>{grade.icon}</div>

        <h2 style={{ fontSize:"32px", fontWeight:"900", color:grade.color, marginBottom:"6px" }}>{grade.label}</h2>

        {name && <p style={{ fontSize:"22px", color:"#78350F", fontWeight:"700", marginBottom:"4px" }}>{name}，辛苦了！</p>}

        <p style={{ ...S.muted, marginBottom:"22px" }}>{grade.msg}</p>


        <div style={{ background:"linear-gradient(135deg,#FEF3C7,#FDE68A)", borderRadius:"22px", padding:"22px", marginBottom:"20px" }}>

          <p style={{ ...S.muted, marginBottom:"8px" }}>🧠 健康防詐金頭腦</p>

          <p style={{ fontSize:"56px", fontWeight:"900", color:"#D97706", lineHeight:1, marginBottom:"4px" }}>

            {correct} <span style={{ fontSize:"26px" }}>/ {total}</span>

          </p>

          <p style={{ fontSize:"24px", color:"#92400E", fontWeight:"700", marginBottom:"14px" }}>答對率 {pct}%</p>

          <div style={{ display:"flex", justifyContent:"center", gap:"10px", flexWrap:"wrap", marginBottom:"14px" }}>

            {scores.map((ok,i) => (

              <div key={i} style={{ width:"42px", height:"42px", borderRadius:"50%", background:ok?"#D1FAE5":"#FEE2E2", border:`3px solid ${ok?"#10B981":"#EF4444"}`, display:"flex", alignItems:"center", justifyContent:"center", fontSize:"18px", fontWeight:"900", color:ok?"#065F46":"#991B1B" }}>

                {ok?"✓":"✗"}

              </div>

            ))}

          </div>

          {tts.supported && (

            <button onClick={() => tts.speak(`您答對了${correct}題，答對率${pct}%，${grade.label}`)}

              style={S.btn({ background:"#F59E0B", color:"#fff", padding:"10px 22px", fontSize:"17px" })}>

              🔊 再聽一次成績

            </button>

          )}

        </div>


        <div style={{ display:"flex", flexDirection:"column", gap:"12px" }}>

          <button onClick={() => { tts.stop(); onRestart(true); }}

            style={S.btn({ width:"100%", height:"68px", background:"linear-gradient(135deg,#F59E0B,#EF4444)", color:"#fff", fontSize:"22px", boxShadow:"0 6px 20px rgba(245,158,11,0.40)" })}>

            🔀 再玩一次（重新抽題）

          </button>

          <button onClick={() => { tts.stop(); onRestart(false); }}

            style={S.btn({ width:"100%", height:"62px", background:"#FEF3C7", color:"#92400E", border:"2px solid #F59E0B", fontSize:"21px" })}>

            🏠 回到首頁

          </button>

        </div>

      </div>

    </div>

  );

}

function App() {

  const [page, setPage] = useState("home");

  const [playerName, setPlayerName] = useState("");

  const [questions, setQuestions] = useState([]);

  const [finalScores, setFinalScores] = useState([]);


  function handleStart(name) { setPlayerName(name); setQuestions(pickQ()); setPage("quiz"); }

  function handleFinish(s) { setFinalScores(s); setPage("result"); }

  function handleRestart(same) {

    if (same) { setQuestions(pickQ()); setFinalScores([]); setPage("quiz"); }

    else setPage("home");

  }


  if (page === "home") return <HomePage onStart={handleStart} />;

  if (page === "quiz") return <QuizPage name={playerName} questions={questions} onFinish={handleFinish} />;

  return <ResultPage name={playerName} scores={finalScores} onRestart={handleRestart} />;

}


ReactDOM.createRoot(document.getElementById("root")).render(<App />);
